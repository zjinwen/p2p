java 客户端 服务端


#运行server

java -jar server.jar

#运行client

java -jar client.jar





配置文件
config.txt


#客户端连接ip地址
config.ip=127.0.0.1

#客户端连接端口，服务端监听端口

config.port=20009

#发送文件目录

config.srcDir=D:\\del\\source\\s

#接收文件目录

config.destDir=D:\\del\\source\\d

#校验成功后删除源文件

config.delAfterSuccess=false
